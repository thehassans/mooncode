# Backend deployment
cd backend
npm ci --production
pm2 restart mooncode-backend || pm2 start src/index.js --name mooncode-backend

# Frontend deployment
cd ../frontend
npm ci
npm run build

# Set permissions (optional)
cd ..
find . -type f -exec chmod 644 {} \;
find . -type d -exec chmod 755 {} \;

echo "Deployment complete at $(date)"
